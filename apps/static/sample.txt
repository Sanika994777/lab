public class SendEmailExample {

    public String message {get; set;}
    public String toEmail {get; set;}
    public Blob fileBody {get; set;}
    public String fileName {get; set;}

    /* Email Validation Method*/
    public static Boolean isValidEmail(String email) {
        if(String.isBlank(email)) {
            return false;
        }
        String emailRegex ='^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\\.[A-Za-z]{2,}$';
        Pattern pattern = Pattern.compile(emailRegex);
        Matcher matcher = pattern.matcher(email);
        return matcher.matches();
    }

    /* Send Email Without Attachment */
    public static void sendEmail() {
        String recipientEmail = 'js8chillin80@gmail.com';
        
        if(!isValidEmail(recipientEmail)) {
            System.debug('Invalid Email Address');
            return;
        }
        
        Messaging.SingleEmailMessage mail = new Messaging.SingleEmailMessage();

        mail.setToAddresses(new String[] {recipientEmail});
        mail.setSubject('Test Email from Salesforce Apex');
        mail.setPlainTextBody('Hello, this is a test email sent using Apex.');

        Messaging.SendEmailResult[] results =
            Messaging.sendEmail(
                new Messaging.SingleEmailMessage[] {mail}
            );

        if(results[0].isSuccess()) {
            System.debug('Email Sent Successfully');
        }
        else {
            System.debug(
                'Email Failed: ' +
                results[0].getErrors()[0].getMessage()
            );
        }
    }

    /* Send Email With Attachment */
    public static void sendEmailWithAttachment() {

        String recipientEmail = 'js8chillin80@gmail.com';

        if(!isValidEmail(recipientEmail)) {
            System.debug('Invalid Email Address');
            return;
        }

        Messaging.SingleEmailMessage mail =
            new Messaging.SingleEmailMessage();

        mail.setToAddresses(new String[] {recipientEmail});
        mail.setSubject('Email with Attachment from Salesforce Apex');
        mail.setPlainTextBody('Please find the attachment.');

        Messaging.EmailFileAttachment attachment = new Messaging.EmailFileAttachment();

        attachment.setFileName('TestFile.txt');
        attachment.setBody(
            Blob.valueOf('This is sample file content')
        );
        attachment.setContentType('text/plain');

        mail.setFileAttachments(
            new Messaging.EmailFileAttachment[] {attachment}
        );

        Messaging.SendEmailResult[] results =
            Messaging.sendEmail(
                new Messaging.SingleEmailMessage[] {mail}
            );

        if(results[0].isSuccess()) {
            System.debug('Email Sent Successfully');
        }
        else {
            System.debug(
                'Email Failed: ' +
                results[0].getErrors()[0].getMessage()
            );
        }
    }

    /*Send Email From Visualforce Page*/
    public PageReference sendDynamicEmail() {

        try {
            // Check Blank Email
            if(String.isBlank(toEmail)) {
                message = 'Please enter email address.';
                return null;
            }

            // Validate Email Format
            if(!isValidEmail(toEmail)) {
                message = 'Invalid email address format.';
                return null;
            }

            Messaging.SingleEmailMessage mail =
                new Messaging.SingleEmailMessage();

            mail.setToAddresses(new String[] {toEmail});
            mail.setSubject('Email from Visualforce');
            mail.setPlainTextBody(
                'File attached from UI.'
            );

            // Attachment Handling
            if(fileBody != null) {
                Messaging.EmailFileAttachment attachment =
                    new Messaging.EmailFileAttachment();

                attachment.setFileName(fileName);
                attachment.setBody(fileBody);

                attachment.setContentType(
                    'application/octet-stream'
                );

                mail.setFileAttachments(
                    new Messaging.EmailFileAttachment[] {
                        attachment
                    }
                );
            }

            Messaging.SendEmailResult[] results =
                Messaging.sendEmail(
                    new Messaging.SingleEmailMessage[] {mail}
                );

            if(results[0].isSuccess()) {
                message = 'Email sent successfully!';
            }
            else {
                message =
                    'Email failed: ' +
                    results[0].getErrors()[0].getMessage();
            }
        }
        catch(Exception e) {
            message = 'Error: ' + e.getMessage();
        }
        return null;
    }
}

public class StudentController {
    public Student__c student { get; set; }
    public List<Student__c> studentList { get; set; }
    public String currentView { get; set; }
    public String validationErrors { get; set; }

    public StudentController() {
        student = new Student__c();
        currentView = 'menu';
        validationErrors = '';
    }

    public void showAdd() {
        student = new Student__c();
        validationErrors = '';
        currentView = 'add';
    }

    public void showEdit() {
        student = new Student__c();
        validationErrors = '';
        currentView = 'edit';
    }

    public void showDelete() {
        student = new Student__c();
        validationErrors = '';
        currentView = 'delete';
    }

    public void showView() {
        loadStudents();
        currentView = 'view';
    }

    public void backToMenu() {
        currentView = 'menu';
        validationErrors = '';
    }

    public void loadStudents() {
        studentList = [SELECT Id, Name, Roll_no__c, Mark__c FROM Student__c ORDER BY Roll_no__c];
    }

    // Called on field blur via actionFunction for real-time validation
    public void validateFields() {
        validationErrors = '';
        List<String> errors = new List<String>();

        if (currentView == 'add' || currentView == 'edit') {
            // Name validation
            if (String.isBlank(student.Name)) {
                errors.add('Name cannot be blank.');
            } else if (!Pattern.matches('[a-zA-Z0-9 ]+', student.Name)) {
                errors.add('Name cannot contain special characters.');
            }

            // Roll No validation
            if (student.Roll_no__c == null) {
                errors.add('Roll Number cannot be blank.');
            } else if (student.Roll_no__c <= 0) {
                errors.add('Roll Number must be a positive number.');
            } else {
                // Uniqueness check (exclude current record if editing)
                List<Student__c> existing = [
                    SELECT Id FROM Student__c
                    WHERE Roll_no__c = :student.Roll_no__c
                    AND Id != :student.Id
                ];
                if (!existing.isEmpty()) {
                    errors.add('Roll Number must be unique.');
                }
            }

            // Marks validation
            if (student.Mark__c == null) {
                errors.add('Marks cannot be blank.');
            } else if (student.Mark__c < 0) {
                errors.add('Marks must be a positive number.');
            } else if (student.Mark__c >= 100) {
                errors.add('Marks must be less than 100.');
            }
        }

        validationErrors = String.join(errors, '|');
    }

    public void saveStudent() {
        validateFields();
        if (String.isNotBlank(validationErrors)) return;

        try {
            insert student;
            ApexPages.addMessage(new ApexPages.Message(
                ApexPages.Severity.CONFIRM, 'Student added successfully!'));
            student = new Student__c();
            validationErrors = '';
            currentView = 'menu';
        } catch (Exception e) {
            ApexPages.addMessage(new ApexPages.Message(
                ApexPages.Severity.ERROR, 'Error: ' + e.getMessage()));
        }
    }

    public void loadStudentForEdit() {
        if (student.Roll_no__c != null) {
            List<Student__c> found = [
                SELECT Id, Name, Roll_no__c, Mark__c
                FROM Student__c
                WHERE Roll_no__c = :student.Roll_no__c
            ];
            if (!found.isEmpty()) {
                student = found[0];
                currentView = 'editForm';
                validationErrors = '';
            } else {
                ApexPages.addMessage(new ApexPages.Message(
                    ApexPages.Severity.ERROR, 'No student found with that Roll Number.'));
            }
        }
    }

    public void updateStudent() {
        validateFields();
        if (String.isNotBlank(validationErrors)) return;

        try {
            update student;
            ApexPages.addMessage(new ApexPages.Message(
                ApexPages.Severity.CONFIRM, 'Student updated successfully!'));
            student = new Student__c();
            validationErrors = '';
            currentView = 'menu';
        } catch (Exception e) {
            ApexPages.addMessage(new ApexPages.Message(
                ApexPages.Severity.ERROR, 'Error: ' + e.getMessage()));
        }
    }

    public void loadStudentForDelete() {
        if (student.Roll_no__c != null) {
            List<Student__c> found = [
                SELECT Id, Name, Roll_no__c, Mark__c
                FROM Student__c
                WHERE Roll_no__c = :student.Roll_no__c
            ];
            if (!found.isEmpty()) {
                student = found[0];
                currentView = 'deleteConfirm';
            } else {
                ApexPages.addMessage(new ApexPages.Message(
                    ApexPages.Severity.ERROR, 'No student found with that Roll Number.'));
            }
        }
    }

    public void deleteStudent() {
        try {
            delete student;
            ApexPages.addMessage(new ApexPages.Message(

public class EmployeeController {

    public Employee__c employee { get; set; }
    public List<Employee__c> employeeList { get; set; }

    public String currentView { get; set; }
    public String validationErrors { get; set; }

    public EmployeeController() {
        employee = new Employee__c();
        currentView = 'menu';
        validationErrors = '';
        loadEmployees();
    }

    public void loadEmployees() {
        employeeList = [
            SELECT Id,
                   Name,
                   Employee_Id__c,
                   Email_Address__c,
                   Joining_Date__c,
                   Salary__c
            FROM Employee__c
            ORDER BY Name
        ];
    }

    public PageReference showAdd() {
        employee = new Employee__c();
        validationErrors = '';
        currentView = 'add';
        return null;
    }

    public PageReference showEdit() {
        employee = new Employee__c();
        validationErrors = '';
        currentView = 'edit';
        return null;
    }

    public PageReference showDelete() {
        employee = new Employee__c();
        validationErrors = '';
        currentView = 'delete';
        return null;
    }

    public PageReference showView() {
        loadEmployees();
        currentView = 'view';
        return null;
    }

    public PageReference backToMenu() {
        employee = new Employee__c();
        validationErrors = '';
        currentView = 'menu';
        return null;
    }

    public void validateFields() {

        List<String> errors = new List<String>();

        if (currentView == 'add' || currentView == 'editForm') {

            if (String.isBlank(employee.Name)) {
                errors.add('Employee Name cannot be blank.');
            }

            if (employee.Employee_Id__c == null) {
                errors.add('Employee ID cannot be blank.');
            } 
            else if (employee.Employee_Id__c <= 0) {
                errors.add('Employee ID must be positive.');
            }

            if (String.isBlank(employee.Email_Address__c)) {
                errors.add('Email Address cannot be blank.');
            }

            if (employee.Joining_Date__c == null) {
                errors.add('Joining Date cannot be blank.');
            }

            if (employee.Salary__c == null) {
                errors.add('Salary cannot be blank.');
            } 
            else if (employee.Salary__c < 0) {
                errors.add('Salary cannot be negative.');
            }
        }

        validationErrors = String.join(errors, '|');
    }

    public void saveEmployee() {

        validateFields();

        if (String.isNotBlank(validationErrors)) {
            return;
        }

        try {

            insert employee;

            ApexPages.addMessage(
                new ApexPages.Message(
                    ApexPages.Severity.CONFIRM,
                    'Employee added successfully!'
                )
            );

            employee = new Employee__c();
            validationErrors = '';
            currentView = 'menu';

            loadEmployees();

        } catch (Exception e) {

            ApexPages.addMessage(
                new ApexPages.Message(
                    ApexPages.Severity.ERROR,
                    'Error: ' + e.getMessage()
                )
            );
        }
    }

    public void loadEmployeeForEdit() {

        if (employee.Employee_Id__c != null) {

            List<Employee__c> found = [

                SELECT Id,
                       Name,
                       Employee_Id__c,
                       Email_Address__c,
                       Joining_Date__c,
                       Salary__c

                FROM Employee__c

                WHERE Employee_Id__c = :employee.Employee_Id__c

                LIMIT 1
            ];

            if (!found.isEmpty()) {

                employee = found[0];
                currentView = 'editForm';
                validationErrors = '';

            } else {

                ApexPages.addMessage(
                    new ApexPages.Message(
                        ApexPages.Severity.ERROR,
                        'No employee found with that Employee ID.'
                    )
                );
            }
        }
    }

    public void updateEmployee() {

        validateFields();

        if (String.isNotBlank(validationErrors)) {
            return;
        }

        try {

            update employee;

            ApexPages.addMessage(
                new ApexPages.Message(
                    ApexPages.Severity.CONFIRM,
                    'Employee updated successfully!'
                )
            );

            employee = new Employee__c();
            validationErrors = '';
            currentView = 'menu';

            loadEmployees();

        } catch (Exception e) {

            ApexPages.addMessage(
                new ApexPages.Message(
                    ApexPages.Severity.ERROR,
                    'Error: ' + e.getMessage()
                )
            );
        }
    }

    public void loadEmployeeForDelete() {

        if (employee.Employee_Id__c != null) {

            List<Employee__c> found = [

                SELECT Id,
                       Name,
                       Employee_Id__c,
                       Email_Address__c,
                       Joining_Date__c,
                       Salary__c

                FROM Employee__c

                WHERE Employee_Id__c = :employee.Employee_Id__c

                LIMIT 1
            ];

            if (!found.isEmpty()) {

                employee = found[0];
                currentView = 'deleteConfirm';

            } else {

                ApexPages.addMessage(
                    new ApexPages.Message(
                        ApexPages.Severity.ERROR,
                        'No employee found with that Employee ID.'
                    )
                );
            }
        }
    }

    public void deleteEmployee() {

        try {

            delete employee;

            ApexPages.addMessage(
                new ApexPages.Message(
                    ApexPages.Severity.CONFIRM,
                    'Employee deleted successfully!'
                )
            );

            employee = new Employee__c();
            currentView = 'menu';

            loadEmployees();

        } catch (Exception e) {

            ApexPages.addMessage(
                new ApexPages.Message(
                    ApexPages.Severity.ERROR,
                    'Error: ' + e.getMessage()
                )
            );
        }
    }
}
